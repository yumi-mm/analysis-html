# WebApp 注册首充分析（2026-06-01）

> 口径依据：`liaoyu-data-analysis` skill  
> 分析对象：**仅** 2026 年 6 月 1 日在 WebApp 端注册的用户（注册后 7 天内首充/充值）  
> 库：`chamet_data`（ODPS / MaxCompute）

---

## 口径说明

| 项 | 约定 |
|---|---|
| WebApp 注册 | `t_user_status.type=10`（注册包），`value IN (36, 40)` |
| 注册日 | `dim_t_user_info.create_time` 日期 = **`2026-06-01`**（仅 6 月 1 日，不含 6/2～6/7） |
| 7 天内充值 | 首充时间 ≥ 注册时间，且与注册日相差 ≤ 6 天（共 7 个自然日，含注册当天） |
| 充值认定（②） | `dwd_finance_recharge_i_d` |
| WebApp 端首充（③及分布） | 全渠道订单中**生涯首笔**充值，且 `app_type IN (36, 40)` |
| 充值场景 | 渠道订单表 `source`（1 视频聊 / 2 直播间 / 3 私聊 / 4 群聊 / 5 朋友圈 / 6 私照等） |
| 第 N 天 | 注册日起第 N 个自然日（第 1 天 = 注册当天，`days_after_reg = 0`） |
| 第 1 天时段 | 注册时刻 → 首充时刻的**间隔小时数**（非当天 0～23 点） |
| 金额单位 | `recharge_money` 一般为美元 |
| 分区 | 充值 `ds` 扫描：`20260601`～`20260607`（6/1 注册 + 7 天）；`t_user_status.ds='20260602'`（6/1 注册日 T+1 快照） |

**ODPS 注意**：每条 SQL 须完整执行（含 `WITH`）；CTE 不能跨多条语句复用。若分步跑，可用文末临时表方案。

## ① 2026-06-01 WebApp 注册用户数

```sql
WITH params AS (
    SELECT
        '2026-06-01' AS reg_date,
        '20260602'     AS snapshot_ds   -- 6/1 注册日 T+1 快照
)
SELECT COUNT(DISTINCT u.userid) AS webapp_reg_user_cnt
FROM chamet_data.dim_t_user_info u
INNER JOIN chamet_data.t_user_status s
    ON  u.userid = s.userid
    AND s.ds    = (SELECT snapshot_ds FROM params LIMIT 1)
    AND s.type  = 10                              -- USERSTATUS_REG_APPTYPE
    AND CAST(s.value AS BIGINT) IN (36, 40)       -- CHAMET_WEBAPP / CHAMET_WEBAPP_PC
WHERE SUBSTRING(u.create_time, 1, 10) = (SELECT reg_date FROM params LIMIT 1)
  AND u.userid <> 1;
```

## ③ WebApp 端首充分布

> 人群：**仅 2026-06-01** WebApp 注册，生涯首充在 WebApp（`app_type` 36/40），且落在注册后 7 天内。  
> 以下 3.1 / 3.2 / 3.3 **各自独立**，须整条复制执行。

### 3.1 充值场景分布

```sql
WITH params AS (
    SELECT
        '2026-06-01' AS reg_date,
        '20260602'     AS snapshot_ds
),
webapp_reg AS (
    SELECT
        u.userid,
        u.create_time AS reg_time
    FROM chamet_data.dim_t_user_info u
    INNER JOIN chamet_data.t_user_status s
        ON  u.userid = s.userid
        AND s.ds    = (SELECT snapshot_ds FROM params LIMIT 1)
        AND s.type  = 10
        AND CAST(s.value AS BIGINT) IN (36, 40)
    WHERE SUBSTRING(u.create_time, 1, 10) = (SELECT reg_date FROM params LIMIT 1)
      AND u.userid <> 1
),
all_recharge_orders AS (
    SELECT userid, pay_time AS recharge_time, CAST(recharge_money AS DOUBLE) AS price,
           source, app_type, order_id
    FROM chamet_data.t_googlepay_recharge
    WHERE recharge_status = 1 AND order_status = 1 AND app_type IN (36, 40)
    UNION ALL
    SELECT userid, pay_time, CAST(recharge_money AS DOUBLE), source, app_type, order_id
    FROM chamet_data.t_cpay_recharge
    WHERE recharge_status = 1 AND order_status = 1 AND app_type IN (36, 40)
    UNION ALL
    SELECT userid, pay_time, CAST(recharge_money AS DOUBLE), source, app_type, order_id
    FROM chamet_data.t_payermax_recharge
    WHERE recharge_status = 1 AND order_status = 1 AND app_type IN (36, 40)
    UNION ALL
    SELECT userid, pay_time, CAST(recharge_money AS DOUBLE), source, app_type, order_id
    FROM chamet_data.t_dlocal_recharge
    WHERE recharge_status = 1 AND order_status = 1 AND app_type IN (36, 40)
    UNION ALL
    SELECT userid, pay_time, CAST(recharge_money AS DOUBLE), source, app_type, order_id
    FROM chamet_data.t_livpay_recharge
    WHERE recharge_status = 1 AND order_status = 1 AND app_type IN (36, 40)
),
first_recharge AS (
    SELECT
        o.userid,
        o.recharge_time,
        o.price,
        o.source,
        ROW_NUMBER() OVER (PARTITION BY o.userid ORDER BY o.recharge_time ASC, o.order_id ASC) AS rn
    FROM all_recharge_orders o
),
target_users AS (
    SELECT
        w.userid,
        f.recharge_time,
        f.price,
        f.source
    FROM webapp_reg w
    INNER JOIN first_recharge f
        ON w.userid = f.userid AND f.rn = 1
    WHERE f.recharge_time >= w.reg_time
      AND DATEDIFF(
              TO_DATE(SUBSTRING(f.recharge_time, 1, 10), 'yyyy-MM-dd'),
              TO_DATE(SUBSTRING(w.reg_time,       1, 10), 'yyyy-MM-dd'),
              'dd'
          ) BETWEEN 0 AND 6
)
SELECT
    source,
    CASE source
        WHEN 1 THEN '视频聊'
        WHEN 2 THEN '直播间'
        WHEN 3 THEN '私聊'
        WHEN 4 THEN '群聊'
        WHEN 5 THEN '朋友圈/随拍'
        WHEN 6 THEN '私照/视频文件/见面'
        ELSE CONCAT('其它(', CAST(source AS STRING), ')')
    END AS source_name,
    COUNT(DISTINCT userid) AS user_cnt,
    ROUND(COUNT(DISTINCT userid) * 100.0 / SUM(COUNT(DISTINCT userid)) OVER (), 2) AS user_pct
FROM target_users
GROUP BY source
ORDER BY user_cnt DESC;
```

### 3.2 充值金额分布

> 0～5 美元按 1 美元一档拆分；5 以上保留原粗分桶。

```sql
    FROM chamet_data.t_livpay_recharge

),
first_recharge AS (
    SELECT
        userid,
        recharge_time,
        price,
        ROW_NUMBER() OVER (PARTITION BY userid ORDER BY recharge_time ASC, order_id ASC) AS rn
    FROM all_recharge_orders
),
target_users AS (
    SELECT w.userid, f.price
    FROM webapp_reg w
    INNER JOIN first_recharge f
        ON w.userid = f.userid AND f.rn = 1
    WHERE f.recharge_time >= w.reg_time
      AND DATEDIFF(
              TO_DATE(SUBSTRING(f.recharge_time, 1, 10), 'yyyy-MM-dd'),
              TO_DATE(SUBSTRING(w.reg_time,       1, 10), 'yyyy-MM-dd'),
              'dd'
          ) BETWEEN 0 AND 6

SELECT
    CASE
        WHEN price >= 0  AND price < 1  THEN '0-1'
        WHEN price >= 1  AND price < 2  THEN '1-2'
        WHEN price >= 2  AND price < 3  THEN '2-3'
        WHEN price >= 3  AND price < 4  THEN '3-4'
        WHEN price >= 4  AND price < 5  THEN '4-5'
        WHEN price >= 5  AND price < 10 THEN '5-10'
        WHEN price >= 10 AND price < 20 THEN '10-20'
        WHEN price >= 20 AND price < 50 THEN '20-50'
        ELSE '>=50'
    END AS price_bucket,
    CASE
        WHEN price >= 0  AND price < 1  THEN 1
        WHEN price >= 1  AND price < 2  THEN 2
        WHEN price >= 2  AND price < 3  THEN 3
        WHEN price >= 3  AND price < 4  THEN 4
        WHEN price >= 4  AND price < 5  THEN 5
        WHEN price >= 5  AND price < 10 THEN 6
        WHEN price >= 10 AND price < 20 THEN 7
        WHEN price >= 20 AND price < 50 THEN 8
        ELSE 9
    END AS bucket_order,
    COUNT(DISTINCT userid) AS user_cnt,
    ROUND(AVG(price), 2) AS avg_price,
    ROUND(SUM(price), 2) AS total_price
FROM target_users
GROUP BY
    CASE
        WHEN price >= 0  AND price < 1  THEN '0-1'
        WHEN price >= 1  AND price < 2  THEN '1-2'
        WHEN price >= 2  AND price < 3  THEN '2-3'
        WHEN price >= 3  AND price < 4  THEN '3-4'
        WHEN price >= 4  AND price < 5  THEN '4-5'
        WHEN price >= 5  AND price < 10 THEN '5-10'
        WHEN price >= 10 AND price < 20 THEN '10-20'
        WHEN price >= 20 AND price < 50 THEN '20-50'
        ELSE '>=50'
    END,
    CASE
        WHEN price >= 0  AND price < 1  THEN 1
        WHEN price >= 1  AND price < 2  THEN 2
        WHEN price >= 2  AND price < 3  THEN 3
        WHEN price >= 3  AND price < 4  THEN 4
        WHEN price >= 4  AND price < 5  THEN 5
        WHEN price >= 5  AND price < 10 THEN 6
        WHEN price >= 10 AND price < 20 THEN 7
        WHEN price >= 20 AND price < 50 THEN 8
        ELSE 9
    END
ORDER BY bucket_order;
```

**金额区间（左闭右开）**：`0-1` → `[0,1)`，`1-2` → `[1,2)`，以此类推。

### 3.4 第 1 天各时段首充人数（注册 → 首充间隔小时）

> **第 1 天**：`days_after_reg = 0`（注册当天首充）。  
> **时段**：从用户注册时刻到首充时刻的间隔小时数（`hour_gap=0` 表示 0～1 小时内）。

```sql
WITH params AS (
    SELECT '2026-06-01' AS reg_date, '20260602' AS snapshot_ds
),
webapp_reg AS (
    SELECT u.userid, u.create_time AS reg_time
    FROM chamet_data.dim_t_user_info u
    INNER JOIN chamet_data.t_user_status s
        ON  u.userid = s.userid
        AND s.ds    = (SELECT snapshot_ds FROM params LIMIT 1)
        AND s.type  = 10
        AND CAST(s.value AS BIGINT) IN (36, 40)
    WHERE SUBSTRING(u.create_time, 1, 10) = (SELECT reg_date FROM params LIMIT 1)
      AND u.userid <> 1
),
all_recharge_orders AS (
    SELECT userid, pay_time AS recharge_time, app_type, order_id
    FROM chamet_data.t_googlepay_recharge
    WHERE recharge_status = 1 AND order_status = 1 AND app_type IN (36, 40)
    UNION ALL
    SELECT userid, pay_time, app_type, order_id
    FROM chamet_data.t_cpay_recharge
    WHERE recharge_status = 1 AND order_status = 1 AND app_type IN (36, 40)
    UNION ALL
    SELECT userid, pay_time, app_type, order_id
    FROM chamet_data.t_payermax_recharge
    WHERE recharge_status = 1 AND order_status = 1 AND app_type IN (36, 40)
    UNION ALL
    SELECT userid, pay_time, app_type, order_id
    FROM chamet_data.t_dlocal_recharge
    WHERE recharge_status = 1 AND order_status = 1 AND app_type IN (36, 40)
    UNION ALL
    SELECT userid, pay_time, app_type, order_id
    FROM chamet_data.t_livpay_recharge
    WHERE recharge_status = 1 AND order_status = 1 AND app_type IN (36, 40)
),
first_recharge AS (
    SELECT
        userid,
        recharge_time,
        ROW_NUMBER() OVER (PARTITION BY userid ORDER BY recharge_time ASC, order_id ASC) AS rn
    FROM all_recharge_orders
),
target_users AS (
    SELECT
        w.userid,
        w.reg_time,
        f.recharge_time,
        DATEDIFF(
            TO_DATE(SUBSTRING(f.recharge_time, 1, 10), 'yyyy-MM-dd'),
            TO_DATE(SUBSTRING(w.reg_time,       1, 10), 'yyyy-MM-dd'),
            'dd'
        ) AS days_after_reg,
        DATEDIFF(
            CAST(f.recharge_time AS DATETIME),
            CAST(w.reg_time AS DATETIME),
            'ss'
        ) / 3600.0 AS hours_after_reg
    FROM webapp_reg w
    INNER JOIN first_recharge f
        ON w.userid = f.userid AND f.rn = 1
    WHERE f.recharge_time >= w.reg_time
      AND DATEDIFF(
              TO_DATE(SUBSTRING(f.recharge_time, 1, 10), 'yyyy-MM-dd'),
              TO_DATE(SUBSTRING(w.reg_time,       1, 10), 'yyyy-MM-dd'),
              'dd'
          ) BETWEEN 0 AND 6
),
day1_users AS (
    SELECT userid, reg_time, recharge_time, hours_after_reg
    FROM target_users
    WHERE days_after_reg = 0
)
SELECT
    hour_gap,
    CONCAT(CAST(hour_gap AS STRING), '-', CAST(hour_gap + 1 AS STRING), '小时') AS hour_range,
    COUNT(DISTINCT userid) AS user_cnt,
    ROUND(AVG(hours_after_reg), 2) AS avg_hours,
    ROUND(MIN(hours_after_reg), 2) AS min_hours,
    ROUND(MAX(hours_after_reg), 2) AS max_hours
FROM (
    SELECT
        userid,
        hours_after_reg,
        FLOOR(hours_after_reg) AS hour_gap
    FROM day1_users
) t
GROUP BY hour_gap
ORDER BY hour_gap;
```

若 `CAST(... AS DATETIME)` 报错，将 `hours_after_reg` 改为：

```sql
(UNIX_TIMESTAMP(f.recharge_time) - UNIX_TIMESTAMP(w.reg_time)) / 3600.0 AS hours_after_reg
```

| 位置 | 现状 | 问题 |
|------|------|------|
| 注册 cohort `a` | `dim_user_f_d` 按 `ds`+`create_time` | **未筛 WebApp 注册包**，含 App/iOS 等全部新用户 |
| 首充 `b` | `server_log_data_91` 全量取 `rn=1` | **首充可能是 App 端**，不是 WebApp 首充 |
| 注册日边界 | `create_time > '2026-03-01'` | 会带上 3-02 及之后注册的用户（若 `ds` 只取一天需加上界） |
| 包类型字段 | 未使用 `t_user_status` | skill 规定注册包用 **type=10**，勿用 `dim_user_f_d.app_type` |

### WebApp 改造要点（改 2 处）

1. **注册 cohort**：`INNER JOIN t_user_status`，`type=10` 且 `value IN (36, 40)`，并 `userid <> 1`
2. **首充埋点**：在 `server_log_data_91` 子查询中加 WebApp 过滤  
   `appType` / `apptype` IN (`36`, `40`)（91 号埋点字段为 **`appType`**，建议两个都判）

### 为何埋点版人数会比渠道订单版多（常见原因）

| # | 问题 | 旧附录 B 写法 | 渠道订单版（①②③） | 后果 |
|---|------|---------------|-------------------|------|
| 1 | **注册用户重复** | `dim_user_f_d` 按 `ds` 扫 7 天，同一 `userid` 可出现 7 行 | `dim_t_user_info` 每人 1 行 | `COUNT(*)` / `SUM(is_paid)` **成倍放大** |
| 2 | **未限制「注册后 7 天内」** | 全表 `rn=1` 取生涯 WebApp 首条埋点 | 显式 `days_after_reg BETWEEN 0 AND 6` | 第 8 天以后首充也被算进「付费」 |
| 3 | **首充与注册未先关联** | 埋点子查询独立算 `rn=1`，再 LEFT JOIN | 先 cohort 再筛该用户订单 | 逻辑上未保证「对该注册用户」的首充 |
| 4 | **未要求首充 ≥ 注册时刻** | 无 `recharge_ts >= register_ts` | 有 | 可能出现负 `diff_seconds` 仍算付费 |
| 5 | **埋点 ≠ 成功订单** | 91 埋点可能含下单/失败态 | 渠道表 `recharge_status=1` | 埋点付费人数偏多 |
| 6 | **统计未去重** | `COUNT(*)` / `SUM(is_paid)` | `COUNT(DISTINCT userid)` | 重复行再次放大 |

**结论**：旧附录 B 虽加了 WebApp 的 `appType` 过滤，但**注册去重、7 天窗口、成功态**未与渠道版对齐，人数偏大是预期现象。下方为**修正版**，口径尽量与渠道订单版一致。

```sql
    GROUP BY money

SELECT
    '【1. 总体与时间分布统计】' AS report_type,

    paid_users,
    unpaid_users,
    ROUND(paid_users * 100.0 / total_users, 2) AS pay_rate_percent,
```

## 附录 C：首充用户的首充场景（406 page / source / scene_type）

### 三种场景来源（优先级）

| 优先级 | 来源 | 字段 | 说明 |
|--------|------|------|------|
| 1 | 406 step=1 | `source` → `source_scene` | 充值入口（速配余额不足等） |

### 首充 + 406 场景明细 SQL（6/1 注册 cohort 示例）

```sql
        '2026-06-01' AS reg_date,
        '20260601'   AS log_ds_start,

webapp_reg AS (

first_recharge_91 AS (

        ) AS rn

    INNER JOIN chamet_data.server_log_data_rt l

      AND l.log_id = '91'
      AND l.userid <> '1'
      AND (GET_JSON_OBJECT(l.extra, '$.appType') IN ('36', '40')
           OR GET_JSON_OBJECT(l.extra, '$.apptype') IN ('36', '40'))
      AND UNIX_TIMESTAMP(COALESCE(GET_JSON_OBJECT(l.extra, '$.createTime'), l.ts)) >= w.register_ts

step4_all AS (

        CAST(userid AS BIGINT) AS userid,
        GET_JSON_OBJECT(extra, '$.transaction_id') AS transaction_id,
        GET_JSON_OBJECT(extra, '$.page') AS page,
        CAST(GET_JSON_OBJECT(extra, '$.value') AS DOUBLE) AS amount_usd,

    WHERE rn = 1
),
step1_all AS (

    LEFT JOIN step4_match s4 ON fr.userid = s4.userid
    LEFT JOIN step1_by_tid s1t ON s4.transaction_id = s1t.transaction_id
    LEFT JOIN step1_by_time s1w
        ON s4.transaction_id = s1w.s4_transaction_id AND s1t.transaction_id IS NULL
)
SELECT
    userid,
    register_time,
    first_recharge_time,
    first_amount_usd,
    page,
    page_scene,
    source,
    source_scene,
    scene_type,
    scene_type_name,
    source_91,
    recharge_source_91,
    has_406_step4,
    has_406_source
FROM enriched
ORDER BY first_recharge_time;
```

### 各场景汇总（用户量 / 订单量 / 金额 / ARPPU / 占比）

> 复用上方 CTE 至 `enriched`，将末尾明细 `SELECT` 替换为下列任一段。  
> 金额取 **91 首充 `first_amount_usd`**；每人 1 笔首充，**充值数量** 通常等于用户数量。  
> **总充值占比** = 该场景总金额 ÷ 全部首充总金额 × 100%。

**按 `page_scene`（406 step=4 页面场景）**：

```sql
SELECT
    page_scene AS 场景,
    用户数量,
    订单数量,
    总充值金额_usd,
    ARPPU_usd,
    ROUND(总充值金额_usd * 100.0 / SUM(总充值金额_usd) OVER (), 2) AS 总充值占比_pct
FROM (
    SELECT
        page_scene,
        COUNT(DISTINCT userid) AS 用户数量,
        COUNT(1) AS 订单数量,
        ROUND(SUM(first_amount_usd), 2) AS 总充值金额_usd,
        ROUND(SUM(first_amount_usd) / COUNT(DISTINCT userid), 2) AS ARPPU_usd
    FROM enriched
    WHERE has_406_step4 = 1
    GROUP BY page_scene
) agg
ORDER BY 总充值金额_usd DESC;
```

**按 `source_scene`（406 step=1 充值入口）**：

```sql
SELECT
    source_scene AS 场景,
    用户数量,
    订单数量,
    总充值金额_usd,
    ARPPU_usd,
    ROUND(总充值金额_usd * 100.0 / SUM(总充值金额_usd) OVER (), 2) AS 总充值占比_pct
FROM (
    SELECT
        source_scene,
        COUNT(DISTINCT userid) AS 用户数量,
        COUNT(1) AS 订单数量,
        ROUND(SUM(first_amount_usd), 2) AS 总充值金额_usd,
        ROUND(SUM(first_amount_usd) / COUNT(DISTINCT userid), 2) AS ARPPU_usd
    FROM enriched
    WHERE has_406_source = 1
    GROUP BY source_scene
) agg
ORDER BY 总充值金额_usd DESC;
```

**按 `scene_type_name`（406 step=1 弹窗子场景）**：

```sql
SELECT
    scene_type,
    scene_type_name AS 场景,
    用户数量,
    订单数量,
    总充值金额_usd,
    ARPPU_usd,
    ROUND(总充值金额_usd * 100.0 / SUM(总充值金额_usd) OVER (), 2) AS 总充值占比_pct
FROM (
    SELECT
        scene_type,
        scene_type_name,
        COUNT(DISTINCT userid) AS 用户数量,

        ROUND(SUM(first_amount_usd) / COUNT(DISTINCT userid), 2) AS ARPPU_usd
    FROM enriched
    WHERE scene_type IS NOT NULL

) agg
ORDER BY 总充值金额_usd DESC;
```

**三场景宽表（一次看 page / source / scene_type 合计，可选）**：

```sql
SELECT 维度, 场景, 用户数量, 订单数量, 总充值金额_usd, ARPPU_usd,
       ROUND(总充值金额_usd * 100.0 / SUM(总充值金额_usd) OVER (PARTITION BY 维度), 2) AS 总充值占比_pct
FROM (
    SELECT 'page_scene' AS 维度, page_scene AS 场景,
           COUNT(DISTINCT userid) AS 用户数量, COUNT(1) AS 订单数量,
           ROUND(SUM(first_amount_usd), 2) AS 总充值金额_usd,
           ROUND(SUM(first_amount_usd) / COUNT(DISTINCT userid), 2) AS ARPPU_usd
    FROM enriched WHERE has_406_step4 = 1 GROUP BY page_scene
    UNION ALL
    SELECT 'source_scene', source_scene,
           COUNT(DISTINCT userid), COUNT(1),
           ROUND(SUM(first_amount_usd), 2),
           ROUND(SUM(first_amount_usd) / COUNT(DISTINCT userid), 2)
    FROM enriched WHERE has_406_source = 1 GROUP BY source_scene
    UNION ALL
    SELECT 'scene_type', scene_type_name,
           COUNT(DISTINCT userid), COUNT(1),
           ROUND(SUM(first_amount_usd), 2),
           ROUND(SUM(first_amount_usd) / COUNT(DISTINCT userid), 2)
    FROM enriched WHERE scene_type IS NOT NULL GROUP BY scene_type_name
) agg
ORDER BY 维度, 总充值金额_usd DESC;
```
