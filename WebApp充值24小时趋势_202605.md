# WebApp 充值 24 小时趋势（2026-05-01～05-31）

> 口径依据：`liaoyu-data-analysis` skill  
> 分析对象：2026 年 5 月内在 **WebApp 端充值** 的用户/订单  
> 输出：按一天中的小时 **0～23** 聚合全月数据 → **充值总金额**、**ARPPU**（与趋势图一致）

---

## 为什么「渠道订单版」人数/金额偏少？

| 原因 | 说明 |
|------|------|
| **渠道未全覆盖** | 原 SQL 只 UNION 了 5 张表，缺 `t_huawei_pay_recharge`、`t_codapay_*` 等 |
| **`app_type` 漏标** | WebApp 大量走三方（cpay/payermax/dlocal），订单行 `app_type` 常 **为空或非 36/40**，被 `IN (36,40)` 滤掉 |
| **未带 `ds` 分区** | 部分渠道表按 `ds` 同步，仅筛 `pay_time` 可能扫不全或性能裁剪异常 |
| **`order_status=1` 过严** | 个别渠道成功态只体现在 `recharge_status=1` |
| **与内部报表口径不一致** | WebApp 日表 `rpt_ly_webapp_d_report`、趋势图多基于 **埋点 log_id=91 + appType**，而非渠道 `app_type` |

**建议**：WebApp 专属 24 小时趋势 **优先用下方「埋点主版」**；渠道版作财务对账；用「三口径对账」SQL 比较差异。

---

## 口径说明

| 项 | 约定 |
|---|---|
| 时间范围 | 充值发生时间 `pay_time` 落在 `2026-05-01`～`2026-05-31` |
| WebApp | 渠道订单 `app_type IN (36, 40)`（CHAMET_WEBAPP / CHAMET_WEBAPP_PC） |
| 成功充值 | `recharge_status = 1` 且 `order_status = 1` |
| 小时 | 从 `pay_time` 取小时 `0～23`（**库内时区**，未做 UTC 换算；与图表不一致时可改 `DATEADD`） |
| 金额 | `recharge_money` 为 **美元（USD）**；`total_amount_万_usd` = 金额 ÷ 10000（万美金） |
| ARPPU | **该小时全月充值总金额 ÷ 该小时全月去重付费用户数**（按「小时桶」聚合，非按日平均） |
| 分区 | 渠道表若有 `ds`，扫描 `20260501`～`20260531`；最终以 `pay_time` 日期为准 |

**与图表对应关系**

| 图表 Y 轴 | SQL 字段 |
|-----------|----------|
| 充值金额（万美金） | `total_amount_万_usd` |
| ARPPU（美金） | `arppu_usd` |
| X 轴 小时 0～23 | `hour_of_day` |

## 主查询：埋点主版（推荐，WebApp 24h 趋势）

> **2026-05 用 `server_log_data_rt` + `log_id='91'`**；以 `extra.appType/apptype` 36/40 识别 WebApp。  
> 金额优先 `money`，其次 `rechargeMoney` / `price`；按 `orderId` 去重避免重复埋点。

```sql
-- WebApp 充值 24 小时趋势（埋点 91 主版，2026-05）
WITH params AS (
    SELECT
        '2026-05-01' AS start_date,
        '2026-05-31' AS end_date,
        '20260501'   AS ds_start,
        '20260531'   AS ds_end,
        '91'         AS log_id
),
hours AS (
    SELECT 0 AS hour_of_day UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3
    UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7
    UNION ALL SELECT 8 UNION ALL SELECT 9 UNION ALL SELECT 10 UNION ALL SELECT 11
    UNION ALL SELECT 12 UNION ALL SELECT 13 UNION ALL SELECT 14 UNION ALL SELECT 15
    UNION ALL SELECT 16 UNION ALL SELECT 17 UNION ALL SELECT 18 UNION ALL SELECT 19
    UNION ALL SELECT 20 UNION ALL SELECT 21 UNION ALL SELECT 22 UNION ALL SELECT 23
),
log_raw AS (
    SELECT
        CAST(userid AS BIGINT) AS userid,
        GET_JSON_OBJECT(extra, '$.orderId') AS order_id,
        COALESCE(
            GET_JSON_OBJECT(extra, '$.createTime'),
            GET_JSON_OBJECT(extra, '$.payTime'),
            ts
        ) AS recharge_time_str,
        CAST(COALESCE(
            GET_JSON_OBJECT(extra, '$.money'),
            GET_JSON_OBJECT(extra, '$.rechargeMoney'),
            GET_JSON_OBJECT(extra, '$.price')
        ) AS DOUBLE) AS amount_usd
    FROM chamet_data.server_log_data_rt
    WHERE ds >= (SELECT ds_start FROM params LIMIT 1)
      AND ds <= (SELECT ds_end FROM params LIMIT 1)
      AND log_id = (SELECT log_id FROM params LIMIT 1)
      AND userid <> '1'
      AND (
          GET_JSON_OBJECT(extra, '$.appType') IN ('36', '40')
          OR GET_JSON_OBJECT(extra, '$.apptype') IN ('36', '40')
      )
      AND SUBSTRING(COALESCE(
              GET_JSON_OBJECT(extra, '$.createTime'),
              GET_JSON_OBJECT(extra, '$.payTime'),
              ts
          ), 1, 10) >= (SELECT start_date FROM params LIMIT 1)
      AND SUBSTRING(COALESCE(
              GET_JSON_OBJECT(extra, '$.createTime'),
              GET_JSON_OBJECT(extra, '$.payTime'),
              ts
          ), 1, 10) <= (SELECT end_date FROM params LIMIT 1)
      AND CAST(COALESCE(
              GET_JSON_OBJECT(extra, '$.money'),
              GET_JSON_OBJECT(extra, '$.rechargeMoney'),
              GET_JSON_OBJECT(extra, '$.price')
          ) AS DOUBLE) > 0
),
log_dedup AS (
    SELECT userid, recharge_time_str, amount_usd
    FROM (
        SELECT
            userid,
            recharge_time_str,
            amount_usd,
            ROW_NUMBER() OVER (
                PARTITION BY COALESCE(order_id, CONCAT(CAST(userid AS STRING), recharge_time_str))
                ORDER BY recharge_time_str ASC
            ) AS rn
        FROM log_raw
    ) t
    WHERE rn = 1
),
hourly_agg AS (
    SELECT
        HOUR(CAST(recharge_time_str AS DATETIME)) AS hour_of_day,
        SUM(amount_usd) AS total_amount_usd,
        COUNT(DISTINCT userid) AS paying_user_cnt,
        COUNT(1) AS order_cnt
    FROM log_dedup
    GROUP BY HOUR(CAST(recharge_time_str AS DATETIME))
)
SELECT
    h.hour_of_day,
    CONCAT(LPAD(CAST(h.hour_of_day AS STRING), 2, '0'), ':00~', LPAD(CAST(h.hour_of_day AS STRING), 2, '0'), ':59') AS hour_label,
    ROUND(NVL(a.total_amount_usd, 0), 2) AS total_amount_usd,
    ROUND(NVL(a.total_amount_usd, 0) / 10000.0, 4) AS total_amount_万_usd,
    NVL(a.paying_user_cnt, 0) AS paying_user_cnt,
    NVL(a.order_cnt, 0) AS order_cnt,
    CASE
        WHEN NVL(a.paying_user_cnt, 0) = 0 THEN 0
        ELSE ROUND(a.total_amount_usd / a.paying_user_cnt, 2)
    END AS arppu_usd
FROM hours h
LEFT JOIN hourly_agg a ON h.hour_of_day = a.hour_of_day
ORDER BY h.hour_of_day;
```

## 备选 A：DWD 财务层 + 埋点圈选 WebApp（金额更贴近财务）

> 用 **log_id=91 + appType 36/40** 圈定 WebApp 充值事件，金额取 **`dwd_finance_recharge_i_d.price`**（与数仓财务口径一致）。

```sql
WITH params AS (
    SELECT '2026-05-01' AS start_date, '2026-05-31' AS end_date,
           '20260501' AS ds_start, '20260531' AS ds_end, '91' AS log_id
),
hours AS (
    SELECT 0 AS hour_of_day UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3
    UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7
    UNION ALL SELECT 8 UNION ALL SELECT 9 UNION ALL SELECT 10 UNION ALL SELECT 11
    UNION ALL SELECT 12 UNION ALL SELECT 13 UNION ALL SELECT 14 UNION ALL SELECT 15
    UNION ALL SELECT 16 UNION ALL SELECT 17 UNION ALL SELECT 18 UNION ALL SELECT 19
    UNION ALL SELECT 20 UNION ALL SELECT 21 UNION ALL SELECT 22 UNION ALL SELECT 23
),
webapp_log AS (
    SELECT DISTINCT
        CAST(userid AS BIGINT) AS userid,
        SUBSTRING(COALESCE(GET_JSON_OBJECT(extra, '$.createTime'), ts), 1, 19) AS recharge_minute
    FROM chamet_data.server_log_data_rt
    WHERE ds >= (SELECT ds_start FROM params LIMIT 1)
      AND ds <= (SELECT ds_end FROM params LIMIT 1)
      AND log_id = (SELECT log_id FROM params LIMIT 1)
      AND userid <> '1'
      AND (
          GET_JSON_OBJECT(extra, '$.appType') IN ('36', '40')
          OR GET_JSON_OBJECT(extra, '$.apptype') IN ('36', '40')
      )
),
dwd_webapp AS (
    SELECT
        d.userid,
        d.recharge_time,
        d.price AS amount_usd
    FROM chamet_data.dwd_finance_recharge_i_d d
    INNER JOIN webapp_log w
        ON d.userid = w.userid
        AND SUBSTRING(d.recharge_time, 1, 19) = w.recharge_minute
    WHERE d.ds >= (SELECT ds_start FROM params LIMIT 1)
      AND d.ds <= (SELECT ds_end FROM params LIMIT 1)
      AND SUBSTRING(d.recharge_time, 1, 10) >= (SELECT start_date FROM params LIMIT 1)
      AND SUBSTRING(d.recharge_time, 1, 10) <= (SELECT end_date FROM params LIMIT 1)
),
hourly_agg AS (
    SELECT
        HOUR(CAST(recharge_time AS DATETIME)) AS hour_of_day,
        SUM(amount_usd) AS total_amount_usd,
        COUNT(DISTINCT userid) AS paying_user_cnt
    FROM dwd_webapp
    GROUP BY HOUR(CAST(recharge_time AS DATETIME))
)
SELECT
    h.hour_of_day,
    ROUND(NVL(a.total_amount_usd, 0) / 10000.0, 4) AS total_amount_万_usd,
    NVL(a.paying_user_cnt, 0) AS paying_user_cnt,
    CASE WHEN NVL(a.paying_user_cnt, 0) = 0 THEN 0
         ELSE ROUND(a.total_amount_usd / a.paying_user_cnt, 2) END AS arppu_usd
FROM hours h
LEFT JOIN hourly_agg a ON h.hour_of_day = a.hour_of_day
ORDER BY h.hour_of_day;
```

## 备选 C：渠道窄口径版（原主查询，易偏少）

> 仅当需要与 **渠道财务订单严格一致** 时使用；WebApp 趋势图通常 **不是** 此口径。

```sql
-- WebApp 充值 24 小时趋势（2026-05 全月）
WITH params AS (
    SELECT
        '2026-05-01' AS start_date,
        '2026-05-31' AS end_date,

all_recharge AS (
    SELECT userid, pay_time AS recharge_time, CAST(recharge_money AS DOUBLE) AS amount_usd, 'googlepay' AS pay_channel

    WHERE recharge_status = 1 AND app_type IN (36, 40)

      AND SUBSTRING(CAST(pay_time AS STRING), 1, 10) <= (SELECT end_date FROM params LIMIT 1)

    SELECT userid, pay_time, CAST(recharge_money AS DOUBLE), 'cpay'
    FROM chamet_data.t_cpay_recharge
    WHERE recharge_status = 1 AND app_type IN (36, 40)

      AND SUBSTRING(CAST(pay_time AS STRING), 1, 10) >= (SELECT start_date FROM params LIMIT 1)

      AND SUBSTRING(CAST(pay_time AS STRING), 1, 10) >= (SELECT start_date FROM params LIMIT 1)
      AND SUBSTRING(CAST(pay_time AS STRING), 1, 10) <= (SELECT end_date FROM params LIMIT 1)

    UNION ALL

    WHERE recharge_status = 1 AND app_type IN (36, 40)

    UNION ALL
    SELECT userid, pay_time, CAST(recharge_money AS DOUBLE), 'livpay'

    WHERE recharge_status = 1 AND app_type IN (36, 40)
      AND (ds IS NULL OR (ds >= (SELECT ds_start FROM params LIMIT 1) AND ds <= (SELECT ds_end FROM params LIMIT 1)))
      AND SUBSTRING(CAST(pay_time AS STRING), 1, 10) >= (SELECT start_date FROM params LIMIT 1)
      AND SUBSTRING(CAST(pay_time AS STRING), 1, 10) <= (SELECT end_date FROM params LIMIT 1)

    UNION ALL
    SELECT userid, pay_time, CAST(recharge_money AS DOUBLE), 'huawei'
    FROM chamet_data.t_huawei_pay_recharge
    WHERE recharge_status = 1 AND app_type IN (36, 40)
      AND (ds IS NULL OR (ds >= (SELECT ds_start FROM params LIMIT 1) AND ds <= (SELECT ds_end FROM params LIMIT 1)))
      AND SUBSTRING(CAST(pay_time AS STRING), 1, 10) >= (SELECT start_date FROM params LIMIT 1)
      AND SUBSTRING(CAST(pay_time AS STRING), 1, 10) <= (SELECT end_date FROM params LIMIT 1)
),
hourly_agg AS (
    SELECT
        HOUR(CAST(recharge_time AS DATETIME)) AS hour_of_day,
        SUM(amount_usd) AS total_amount_usd,
        COUNT(DISTINCT userid) AS paying_user_cnt,
        COUNT(1) AS order_cnt
    FROM all_recharge
    WHERE recharge_time IS NOT NULL
    GROUP BY HOUR(CAST(recharge_time AS DATETIME))

SELECT
    h.hour_of_day,
    CONCAT(LPAD(CAST(h.hour_of_day AS STRING), 2, '0'), ':00~', LPAD(CAST(h.hour_of_day AS STRING), 2, '0'), ':59') AS hour_label,

    NVL(a.order_cnt, 0) AS order_cnt,
    CASE
        WHEN NVL(a.paying_user_cnt, 0) = 0 THEN 0
        ELSE ROUND(a.total_amount_usd / a.paying_user_cnt, 2)
    END AS arppu_usd
FROM hours h
LEFT JOIN hourly_agg a
```

## 补充：按日 × 小时明细（可选，用于验数 / 画热力图）

```sql
WITH params AS (
    SELECT '2026-05-01' AS start_date, '2026-05-31' AS end_date
),
all_recharge AS (
    SELECT userid, pay_time AS recharge_time, CAST(recharge_money AS DOUBLE) AS amount_usd
    FROM chamet_data.t_cpay_recharge
    WHERE recharge_status = 1 AND order_status = 1 AND app_type IN (36, 40)
      AND SUBSTRING(CAST(pay_time AS STRING), 1, 10) >= (SELECT start_date FROM params LIMIT 1)
      AND SUBSTRING(CAST(pay_time AS STRING), 1, 10) <= (SELECT end_date FROM params LIMIT 1)
    -- 其余渠道 UNION ALL 同主查询
)
SELECT
    SUBSTRING(CAST(recharge_time AS STRING), 1, 10) AS recharge_date,
    HOUR(CAST(recharge_time AS DATETIME)) AS hour_of_day,
    SUM(amount_usd) AS total_amount_usd,
    COUNT(DISTINCT userid) AS paying_user_cnt,
    ROUND(SUM(amount_usd) / COUNT(DISTINCT userid), 2) AS arppu_usd
FROM all_recharge
GROUP BY
    SUBSTRING(CAST(recharge_time AS STRING), 1, 10),
    HOUR(CAST(recharge_time AS DATETIME))
ORDER BY recharge_date, hour_of_day;
```

## 三口径对账（5 月全月合计，先跑这个）

```sql
-- 1) 渠道窄口径（app_type=36/40，原逻辑）
SELECT 'channel_app_type_36_40' AS口径,
       COUNT(1) AS order_cnt,
       COUNT(DISTINCT userid) AS user_cnt,
       ROUND(SUM(CAST(recharge_money AS DOUBLE)), 2) AS amount_usd
FROM (
    SELECT userid, recharge_money FROM chamet_data.t_cpay_recharge
    WHERE recharge_status = 1 AND app_type IN (36, 40)
      AND SUBSTRING(CAST(pay_time AS STRING), 1, 10) BETWEEN '2026-05-01' AND '2026-05-31'
    UNION ALL
    SELECT userid, recharge_money FROM chamet_data.t_payermax_recharge
    WHERE recharge_status = 1 AND app_type IN (36, 40)
      AND SUBSTRING(CAST(pay_time AS STRING), 1, 10) BETWEEN '2026-05-01' AND '2026-05-31'
    UNION ALL
    SELECT userid, recharge_money FROM chamet_data.t_dlocal_recharge
    WHERE recharge_status = 1 AND app_type IN (36, 40)
      AND SUBSTRING(CAST(pay_time AS STRING), 1, 10) BETWEEN '2026-05-01' AND '2026-05-31'
    UNION ALL
    SELECT userid, recharge_money FROM chamet_data.t_livpay_recharge
    WHERE recharge_status = 1 AND app_type IN (36, 40)
      AND SUBSTRING(CAST(pay_time AS STRING), 1, 10) BETWEEN '2026-05-01' AND '2026-05-31'
) t

UNION ALL

-- 2) 渠道宽口径（5 月成功充值全部，不看 app_type — 看漏标规模）
SELECT 'channel_all_success' AS口径,
       COUNT(1), COUNT(DISTINCT userid),
       ROUND(SUM(CAST(recharge_money AS DOUBLE)), 2)
FROM (
    SELECT userid, recharge_money FROM chamet_data.t_cpay_recharge
    WHERE recharge_status = 1
      AND SUBSTRING(CAST(pay_time AS STRING), 1, 10) BETWEEN '2026-05-01' AND '2026-05-31'
    UNION ALL
    SELECT userid, recharge_money FROM chamet_data.t_payermax_recharge
    WHERE recharge_status = 1
      AND SUBSTRING(CAST(pay_time AS STRING), 1, 10) BETWEEN '2026-05-01' AND '2026-05-31'
    UNION ALL
    SELECT userid, recharge_money FROM chamet_data.t_dlocal_recharge
    WHERE recharge_status = 1
      AND SUBSTRING(CAST(pay_time AS STRING), 1, 10) BETWEEN '2026-05-01' AND '2026-05-31'
    UNION ALL
    SELECT userid, recharge_money FROM chamet_data.t_livpay_recharge
    WHERE recharge_status = 1
      AND SUBSTRING(CAST(pay_time AS STRING), 1, 10) BETWEEN '2026-05-01' AND '2026-05-31'
) t

UNION ALL

-- 3) 埋点 WebApp（log_id=91, appType 36/40）
SELECT 'log91_webapp' AS口径,
       COUNT(1), COUNT(DISTINCT userid),
       ROUND(SUM(CAST(COALESCE(
           GET_JSON_OBJECT(extra, '$.money'),
           GET_JSON_OBJECT(extra, '$.rechargeMoney')
       ) AS DOUBLE)), 2)
FROM chamet_data.server_log_data_rt
WHERE ds >= '20260501' AND ds <= '20260531'
  AND log_id = '91' AND userid <> '1'
  AND (GET_JSON_OBJECT(extra, '$.appType') IN ('36','40')
       OR GET_JSON_OBJECT(extra, '$.apptype') IN ('36','40'))
  AND SUBSTRING(COALESCE(GET_JSON_OBJECT(extra, '$.createTime'), ts), 1, 10)
      BETWEEN '2026-05-01' AND '2026-05-31'

UNION ALL

-- 4) WebApp 日报表（口径可能为人民币，仅作量级参考）
SELECT 'rpt_ly_webapp_d_report' AS口径,
       NULL AS order_cnt,
       SUM(recharge_cnt) AS user_cnt,
       SUM(recharge_sum) AS amount_ref
FROM chamet_data.rpt_ly_webapp_d_report
WHERE ds >= '20260501' AND ds <= '20260531';
```

```sql
-- 5) 看 app_type 漏标：cpay 5 月成功单里 app_type 分布
SELECT
```
